[[User Defined Data Types in C++]]

```cpp
 // Integer
    int age = 21;
    int year = 2023;
    int days = 7;

    // Double
    double price = 10.99;

    // Single Character
    char grade = 'A';
    char initial = 'B';
    char dollar = '$';

    // Boolean
    bool student = false;
    bool power = true;

    // String
    std::string name = "Bro";
    std::string day = "Friday";
    std::string food = "Pizza";

    // Constant
    const double PI = 3.14159;
    double rad = 10;
    double circum = 2 * PI * rad;
    const int LIGHT_SPEED = 299792458;
```
